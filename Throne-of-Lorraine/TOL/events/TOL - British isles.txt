country_event = {
    id = 28015
    title = "EVTNAME28015"
    desc = "EVTDESC28015"
    picture = "crete_revolt"
	fire_only_once = yes

    trigger = {
	    tag = ENG
	    NOT = { has_country_flag = irish_revolt }
		any_pop = { 
		    limit = { 
			    has_pop_culture = irish
	            average_militancy = 4 
			}
        }
		OR = {
		    263 = { owned_by = THIS }
			254 = { owned_by = THIS }
		}
        NOT = { exists = IRR }
    }

    mean_time_to_happen = {
        months = 3
        modifier = {
            factor = 0.5
            average_militancy = 7
        }

        modifier = {
            factor = 0.5
            has_recently_lost_war = yes
        }
    }
	
	option = {
        name = "Well then..."
		set_country_flag = irish_revolt
        any_owned = {
            limit = { is_core = IRE }
            secede_province = IRR
        }
        diplomatic_influence = {
            who = IRR
            value = -400
        }
        casus_belli = {
            target = IRR 
            type = conquest
            months = 12
        }
        war = {
            target = IRR 
            attacker_goal = { casus_belli = conquest }
            defender_goal = { casus_belli = status_quo }
        }
		IRE = country_event = 28018
		random = {
		    chance = 0.2
	        set_country_flag = irish_integration
	    }
        ai_chance = {
            factor = 100
        }
    }
}

	    has_country_flag = irish_integration
	    tag = ENG
		war = no
    }

        days = 20
    }
	
	option = {
        name = "Integrate the Irish culture then."
		add_accepted_culture = irish

		    limit = {
